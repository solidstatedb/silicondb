495  // For each output format that is enabled (OutputGenerator::isEnabled()) we forward
496  // the method call.
497  // We use C++11 variadic templates and perfect forwarding to implement forall() generically,
498  // and split the types of the methods from the arguments passed to allow implicit conversions.
499  template<typename T,class... Ts,class... As>
500  void forall(void (T::*methodPtr)(Ts...),As&&... args)
501  {
502  for (const auto &og : m_outputs)
503  {
504  if (og->isEnabled()) (og.get()->*methodPtr)(std::forward<As>(args)...);
505  }
506  }
